```mermaid
        @startuml DryCleanerProcess

start

:顾客携带衣物到店;
partition 收银员处理 {
    if (是会员吗？) then (是)
        :选择会员信息;
    else (否)
        :注册新会员;
        :充值账户;
    endif
    :创建订单;
    :选择衣物类型、数量、服务;
    :系统计算价格（含折扣/优惠券）;
    :生成预计取衣时间;
    if (账户余额充足？) then (是)
        :扣款;
    else (否)
        :提示充值;
        back :充值账户;
    endif
    :订单状态转为【清洗中】;
}

partition 洗衣处理 {
    :移交洗衣员;
    if (服务类型) then (干洗)
        :执行干洗流程;
    else (水洗)
        :执行水洗流程;
    else (熨烫)
        :执行熨烫流程;
    endif
    :完成处理;
    :订单状态转为【待取走】;
}

partition 取衣处理 {
    :顾客前来取衣;
    if (是本人吗？) then (是)
        :验证身份;
    else (否)
        :记录代取人信息;
    endif
    :标记订单为【已取走】;
    if (是否逾期？) then (是)
        :触发逾期提醒;
    endif
}

stop

@enduml
```
